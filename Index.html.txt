  <audio id="click-sound" src="https://actions.google.com/sounds/v1/cartoon/wood_plank_flicks.ogg"></audio>

  <script>
    let points = 0;

    const storyData = {
      "stories": [
        {
          "title": "Hello Kitty and the Rainbow Unicorn",
          "words": [
            "Hello",
            "Kitty",
            "met",
            "a",
            "sparkly",
            "unicorn",
            "in",
            "Rainbow",
            "Valley"
          ]
        }
      ]
    };

    function loadStory() {
      const story = storyData.stories[0]; // First story
      const storyTextDiv = document.getElementById('story-text');
      const audio = document.getElementById('click-sound');
      storyTextDiv.innerHTML = '';

      story.words.forEach((word) => {
        const span = document.createElement('span');
        span.innerText = word + ' ';
        span.style.cursor = 'pointer';
        span.onclick = () => {
          points += 1;
          document.getElementById('points').innerText = `Points: ${points}`;
          audio.play();
        };
        storyTextDiv.appendChild(span);
      });
    }
  </script>
